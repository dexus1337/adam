267:         float scroll_y,
268:         float window_h,
269:         int& out_display_start,
270:         int& out_display_end,
271:         float& out_top_dummy_h,
272:         float& out_bottom_dummy_h,
273:         heigth_func get_expanded_height
274:     )
275:     {
276:         out_display_start = -1;
277:         out_display_end = -1;
278:         out_top_dummy_h = 0.0f;
279:         out_bottom_dummy_h = 0.0f;
280:         
281:         if (num_total_rows == 0) return;
282: 
283:         float current_y = 0.0f;
284:         int current_idx = 0;
285:         
286:         for (size_t exp_idx : expanded_nodes)
287:         {
288:             if ((int)exp_idx >= num_total_rows) continue;
289:             
290:             int unexpanded_count = (int)exp_idx - current_idx;
291:             if (unexpanded_count > 0)
292:             {
293:                 float unexpanded_h = unexpanded_count * row_height;
294:                 if (out_display_start == -1 && current_y + unexpanded_h >= scroll_y)
295:                 {
296:                     float overshoot = scroll_y - current_y;
297:                     int rows_in = (int)(overshoot / row_height);
298:                     out_display_start = current_idx + rows_in;
299:                 }
300:                 if (out_display_end == -1 && current_y + unexpanded_h >= scroll_y + window_h)
301:                 {
302:                     float overshoot = (scroll_y + window_h) - current_y;
303:                     int rows_in = (int)(overshoot / row_height);
304:                     out_display_end = current_idx + rows_in;
305:                     if (overshoot > rows_in * row_height) out_display_end += 1;
306:                 }
307:                 current_y += unexpanded_h;
308:                 current_idx = (int)exp_idx;
309:             }
310:             
311:             float exp_h = get_expanded_height(exp_idx);
312:             float item_h = row_height + exp_h;
313:             
314:             if (out_display_start == -1 && current_y + item_h >= scroll_y) out_display_start = current_idx;
315:             if (out_display_end == -1 && current_y + item_h >= scroll_y + window_h)
316:             {
317:                 out_display_end = current_idx + 1;
318:             }
319:             
320:             current_y += item_h;
321:             current_idx++;
322:         }
323:         
324:         if (current_idx < num_total_rows)
325:         {
326:             int unexpanded_count = num_total_rows - current_idx;
327:             float unexpanded_h = unexpanded_count * row_height;
328:             if (out_display_start == -1 && current_y + unexpanded_h >= scroll_y)
329:             {
330:                 float overshoot = scroll_y - current_y;
331:                 int rows_in = (int)(overshoot / row_height);
332:                 out_display_start = current_idx + rows_in;
333:             }
334:             if (out_display_end == -1 && current_y + unexpanded_h >= scroll_y + window_h)
335:             {
336:                 float overshoot = (scroll_y + window_h) - current_y;
337:                 int rows_in = (int)(overshoot / row_height);
338:                 out_display_end = current_idx + rows_in;
339:                 if (overshoot > rows_in * row_height) out_display_end += 1;
340:             }
341:             current_y += unexpanded_h;
342:         }
343:         
344:         if (out_display_start == -1) out_display_start = num_total_rows;
345:         if (out_display_end == -1) out_display_end = num_total_rows;
346:         if (out_display_end > num_total_rows) out_display_end = num_total_rows;
347:         if (out_display_start > out_display_end) out_display_start = out_display_end;
348: 
349:         out_top_dummy_h = out_display_start * row_height;
350:         for (size_t exp_idx : expanded_nodes)
351:         {
352:             if ((int)exp_idx < out_display_start) out_top_dummy_h += get_expanded_height(exp_idx);
353:             else break;
354:         }
355:         
356:         out_bottom_dummy_h = (num_total_rows - out_display_end) * row_height;
357:         for (auto it = expanded_nodes.rbegin(); it != expanded_nodes.rend(); ++it)
358:         {
359:             if ((int)*it >= out_display_end && (int)*it < num_total_rows) out_bottom_dummy_h += get_expanded_height(*it);
360:             else if ((int)*it < out_display_end) break;
361:         }
362:     }
